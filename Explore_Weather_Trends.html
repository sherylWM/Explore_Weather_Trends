import matplotlib.pyplot as plt
import numpy as np

localCityYearX = cityTempDF['year']
localCityAvgTempY = cityTempDF['avg_temp']
plt.plot(localCityYearX, localCityAvgTempY, label = "Indianapolis")

globalYearX = globalTempDF['year']
globalTempY = globalTempDF['avg_temp']
plt.plot(globalYearX, globalTempY, label = "Global")

plt.rcParams["figure.figsize"] = (20, 5)
plt.xlabel('Year')
plt.ylabel('Avg. Temp')
plt.legend()
plt.title('Avg Temp in Indianapolis and the Globe by Year')
plt.xticks(np.arange(1740, 2015, 10)) 

plt.show()
